    detector = MTCNN()
    results = detector.detect_faces(pixels)

        flag=1

def get_embedding(model, face_pixels):
    face_pixels = face_pixels.astype('float32')
    mean, std = face_pixels.mean(), face_pixels.std()
    face_pixels = (face_pixels - mean) / std
    samples = expand_dims(face_pixels, axis=0)
    ypred = model.predict(samples)
    return ypred[0]

# A function to extract faces of person from image
def extract_face_test(filename, required_size=(160, 160)):
    image = Image.open(filename)
    image = image.convert('RGB')
    pixels = asarray(image)
    detector = MTCNN()
    results = detector.detect_faces(pixels)
    for i in range(0,len(results)):
        if results[i]['confidence']>0.99:
            x1, y1, width, height = results[i]['box']
            x1, y1 = abs(x1), abs(y1)
            x2, y2 = x1 + width, y1 + height
            face = pixels[y1:y2, x1:x2]
            image = Image.fromarray(face)
            image = image.resize(required_size)
            face_array = asarray(image)
            mf1.append(face_array)
    
    return mf1

# FOR CHECKiNG IF IMAGE IS ROTATED AND THEN ROTAING IMAGE
from PIL import Image, ExifTags

try:
    image=Image.open('h.jpg')
    for orientation in ExifTags.TAGS.keys():
        if ExifTags.TAGS[orientation]=='Orientation':
            break
    exif=dict(image._getexif().items())

    if exif[orientation] == 3:
        image=image.rotate(180, expand=True)
    elif exif[orientation] == 6:
        image=image.rotate(270, expand=True)
    elif exif[orientation] == 8:
        image=image.rotate(90, expand=True)
    image.save('h1.jpg')
    image.close()

except (AttributeError, KeyError, IndexError):
    
    pass

#FOR DATA AUGMENTATION
path = './s'
import os
from keras.preprocessing.image import ImageDataGenerator,img_to_array, load_img
datagen = ImageDataGenerator(
        rotation_range=40,
        #width_shift_range=0.2,
        #height_shift_range=0.2,
        shear_range=0.2,
        brightness_range=[0.2,1.0],
        zoom_range=0.2,
        horizontal_flip=True,
        fill_mode='nearest')
for i in os.listdir(path):
    for j in os.listdir(path+'/'+f'{i}'):
        img = load_img(path+'/'+f'{i}'+'/'+f'{j}')  
        x = img_to_array(img)
        x = x.reshape((1,) + x.shape)
        k=1
        for batch in datagen.flow(x,save_to_dir=path+'/'+f'{i}' ,save_prefix='c', save_format='jpeg'):
            k += 1
            if k > 7:
                break
